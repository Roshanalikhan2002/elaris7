<script>
(function() {
  // ===== Mobile Menu Toggle =====
  const menuToggle = document.getElementById('menu-toggle');
  const mobileMenu = document.getElementById('mobile-menu');
  const menuClose = document.getElementById('menu-close-btn');

  function openMenu() {
    if (!menuToggle || !mobileMenu) return;
    menuToggle.classList.add('is-open');
    mobileMenu.classList.add('open');
    document.body.classList.add('menu-open');
  }

  function closeMenu() {
    if (!menuToggle || !mobileMenu) return;
    menuToggle.classList.remove('is-open');
    mobileMenu.classList.remove('open');
    document.body.classList.remove('menu-open');
  }

  if (menuToggle) menuToggle.addEventListener('click', openMenu);
  if (menuClose) menuClose.addEventListener('click', closeMenu);

  // Mobile Menu filtering is now handled globally in script.js to ensure consistency.
  // This avoids conflicts between two different scripts trying to filter the same elements.


  // Mega Menu links are handled by standard navigation

  // ===== Search Overlay Toggle =====
  const searchTriggers = document.querySelectorAll('.search-trigger');
  const searchOverlay = document.getElementById('search-overlay');
  const searchClose = document.getElementById('search-close');
  const searchInput = document.getElementById('search-input');

  function openSearch(e) {
    if (e) e.preventDefault();
    if (!searchOverlay) return;
    searchOverlay.classList.add('active');
    document.body.classList.add('menu-open');
    setTimeout(() => {
      if (searchInput) searchInput.focus();
    }, 400);
  }

  function closeSearch() {
    if (!searchOverlay) return;
    searchOverlay.classList.remove('active');
    if (!document.getElementById('mobile-menu').classList.contains('open')) {
      document.body.classList.remove('menu-open');
    }
  }

  searchTriggers.forEach(trigger => trigger.addEventListener('click', openSearch));
  if (searchClose) searchClose.addEventListener('click', closeSearch);

  // Esc key to close everything
  document.addEventListener('keydown', (e) => {
    if (e.key === 'Escape') {
      closeSearch();
      closeMenu();
    }
  });

  // ===== Header Scroll Logic =====
  var header = document.querySelector('.site-header');
  if (!header) return;

  var lastScrollY = window.pageYOffset || document.documentElement.scrollTop;
  var hideTimer = null;
  var isHidden = false;
  var threshold = 5;

  function showHeader() {
    if (hideTimer) { clearTimeout(hideTimer); hideTimer = null; }
    if (isHidden) {
      header.classList.remove('header-hidden');
      header.classList.add('header-visible');
      isHidden = false;
    }
  }

  function scheduleHide() {
    if (!hideTimer && !isHidden) {
      hideTimer = setTimeout(function() {
        var currentY = window.pageYOffset || document.documentElement.scrollTop;
        const megaIsHovered = document.querySelector('.nav-item-shop:hover');
        if (!isHidden && currentY > 100 && !megaIsHovered) {
          header.classList.remove('header-visible');
          header.classList.add('header-hidden');
          isHidden = true;
        }
        hideTimer = null;
      }, 2000);
    }
  }

  window.addEventListener('scroll', function() {
    var currentScrollY = window.pageYOffset || document.documentElement.scrollTop;
    if (Math.abs(currentScrollY - lastScrollY) < threshold) return;
    if (currentScrollY <= 20) { showHeader(); } 
    else if (currentScrollY > lastScrollY) { scheduleHide(); } 
    else { showHeader(); }
    lastScrollY = currentScrollY;
  }, { passive: true });
})();
</script>

            const b = this;

            } catch (error) {
                console.error('Cart Addition Error:', error);
                alert('Bag Error: ' + error.message);
                b.innerHTML = originalText;
                b.disabled = false;
            }
        });
    });
